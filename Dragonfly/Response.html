def to_response
  if etag_matches?
    # Not Modified
    [304, cache_headers, []]
  else
    # Success
    [200, success_headers.merge(cache_headers), job.result]
  end
rescue DataStorage::DataNotFound => e
  [404, {"Content-Type" => 'text/plain'}, [e.message]]
end

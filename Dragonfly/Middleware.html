def initialize(app, dragonfly_app_name, path_prefix)
  @app = app
  @endpoint = Rack::Builder.new {
    map path_prefix do
      run Dragonfly[dragonfly_app_name]
    end
  }.to_app
end

def call(env)
  response = @endpoint.call(env)
  if route_not_found?(response)
    @app.call(env)
  else
    response
  end
end

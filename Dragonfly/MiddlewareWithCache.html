def initialize(app, dragonfly_app_name, rack_cache_opts={})
  super(app, dragonfly_app_name)
  @rack_cache_opts = {
    :verbose     => true,
    :metastore   => 'file:/var/cache/rack/meta',
    :entitystore => 'file:/var/cache/rack/body'
  }.merge(rack_cache_opts)
end
